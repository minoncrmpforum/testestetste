  // ОНЛАЙН СЕТЬ (P2P PEERJS)
  function initP2P(role) {
    myRole = role;
    if(role === 'host') {
      myName = document.getElementById('inp-host-name').value.trim() || 'Смотритель';
      const maxP = parseInt(document.getElementById('sel-player-limit').value);
      
      // Сбор конфига бункера хостом
      const disaster = disasters[Math.floor(Math.random() * disasters.length)];
      gameConfig = {
        disaster: disaster,
        condition: bunkerConditions[Math.floor(Math.random() * bunkerConditions.length)],
        energy: [50, 75, 100][Math.floor(Math.random() * 3)],
        maxPlayers: maxP,
        targetSurvivors: Math.max(1, Math.floor(maxP / 2))
      };

      peer = new Peer();
      peer.on('open', (id) => {
        document.getElementById('bunker-code-txt').innerText = id;
        document.getElementById('host-share-zone').style.display = 'block';
        
        let myChar = createCharacterData();
        activePlayers.push({ id: 'Host', name: myName, role: 'host', char: myChar, kicked: false });
        
        renderCharacter(myChar);
        applyWorldConfig(gameConfig);
        checkOxygenStatus();
        
        // Автоматический переход в игру для хоста
        setTimeout(() => navigate('scr-network', 'scr-game'), 4000);
      });

      peer.on('connection', (c) => {
        if(connections.length >= gameConfig.maxPlayers - 1) {
          c.on('open', () => c.send({ type: 'error', msg: 'Бункер переполнен!' }));
          return;
        }
        connections.push(c);
        handleHostTraffic(c);
      });

    } else {
      // КЛИЕНТ
      myName = document.getElementById('inp-client-name').value.trim() || 'Выживший';
      const tId = document.getElementById('inp-target-id').value.trim();
      if(!tId) return alert('Введите код бункера!');

      peer = new Peer();
      peer.on('open', () => {
        conn = peer.connect(tId);
        conn.on('open', () => {
          conn.send({ type: 'join', name: myName });
        });
        handleClientTraffic(conn);
      });
    }
  }

  function handleHostTraffic(c) {
    c.on('data', (pack) => {
      if(pack.type === 'join') {
        let clientChar = createCharacterData();
        activePlayers.push({ id: c.peer, name: pack.name, role: 'client', char: clientChar, kicked: false });
        
        c.send({ type: 'init-game', char: clientChar, config: gameConfig });
        syncAll();
      }
    });
    c.on('close', () => {
      activePlayers = activePlayers.filter(p => p.id !== c.peer);
      syncAll();
    });
  }

  function handleClientTraffic(c) {
    c.on('data', (pack) => {
      if(pack.type === 'init-game') {
        renderCharacter(pack.char);
        applyWorldConfig(pack.config);
        gameConfig = pack.config;
        document.getElementById('my-name-badge').innerText = myName;
        navigate('scr-network', 'scr-game');
      }
      if(pack.type === 'sync') {
        activePlayers = pack.players;
        renderPlayersTerminal();
        checkOxygenStatus();
      }
      if(pack.type === 'error') {
        alert(pack.msg);
        window.location.reload();
      }
      if(pack.type === 'game-end') {
        showGameOver(pack.msg);
      }
    });
  }

  function syncAll() {
    renderPlayersTerminal();
    checkOxygenStatus();
    connections.forEach(c => {
      if(c.open) c.send({ type: 'sync', players: activePlayers });
    });
    
    // Проверка лимита выживших для завершения
    const remaining = activePlayers.filter(p => !p.kicked).length;
    if(remaining === gameConfig.targetSurvivors) {
      let msg = `Шлюзовые двери намертво запечатаны. Выжившие внутри: ` + activePlayers.filter(p => !p.kicked).map(p => p.name).join(', ') + `. Ресурсов хватит для построения нового общества!`;
      connections.forEach(c => c.open && c.send({ type: 'game-end', msg: msg }));
      showGameOver(msg);
    }
  }

  function applyWorldConfig(cfg) {
    document.getElementById('world-disaster-text').innerText = `КАТАСТРОФА: ${cfg.disaster.title} (${cfg.disaster.desc})`;
    document.getElementById('bunker-condition-text').innerText = `Особые условия: ${cfg.condition}`;
    
    // Температура
    document.getElementById('f-txt-temp').innerText = cfg.disaster.temp;
    const tempBox = document.getElementById('flask-temp-box');
    tempBox.className = "flask-container " + cfg.disaster.style;
    document.getElementById('f-liq-temp').style.height = cfg.disaster.style === 'cold' ? '25%' : '85%';

    // Энергия
    document.getElementById('f-txt-power').innerText = cfg.energy + "%";
    document.getElementById('f-liq-power').style.height = cfg.energy + "%";
    document.getElementById('f-liq-power').style.background = "#ffaa00";
  }

  // ДИНАМИЧЕСКИЙ РАСЧЕТ КИСЛОРОДА НА ОСНОВЕ ОСТАВШИХСЯ ЖИВЫХ ИГРОКОВ
  function checkOxygenStatus() {
    // Проверяем только тех игроков, которые ЕЩЕ НЕ изгнаны
    const alivePlayers = activePlayers.filter(p => !p.kicked);
    const hasInfected = alivePlayers.some(p => p.char.isAirPolluter);

    const o2Box = document.getElementById('flask-o2-box');
    const o2Liq = document.getElementById('f-liq-o2');
    const o2Txt = document.getElementById('f-txt-o2');

    if(hasInfected) {
      o2Box.classList.add('danger');
      o2Liq.style.height = "25%";
      o2Txt.innerText = "КРИТИЧЕСКИЙ (ЗАРАЖЕН)";
    } else {
      o2Box.classList.remove('danger');
      o2Liq.style.height = "100%";
      o2Txt.innerText = "НОРМА";
    }
  }

  function renderPlayersTerminal() {
    const box = document.getElementById('players-terminal-list');
    const aliveCount = activePlayers.filter(p => !p.kicked).length;
    document.getElementById('total-surv-count').innerText = `${aliveCount}/${gameConfig ? gameConfig.maxPlayers : activePlayers.length}`;

    box.innerHTML = activePlayers.map(p => `
      <div class="player-row ${p.kicked ? 'exiled' : ''}">
        <span>${p.name} <small style="color:var(--orange); font-size:0.65rem;">[${p.role.toUpperCase()}]</small></span>
        <div>
          ${myRole === 'host' && !p.kicked ? `<button class="btn btn-sm" style="border-color:var(--neon-red); color:var(--neon-red);" onclick="kickPlayer('${p.id}')">Изгнать</button>` : ''}
          ${p.kicked ? `<span style="color:var(--neon-red); font-size:0.75rem;">ИЗГНАН</span>` : `<span style="color:var(--neon-green); font-size:0.75rem;">В БУНКЕРЕ</span>`}
        </div>
      </div>
    `).join('');
  }

  function kickPlayer(id) {
    if(myRole !== 'host') return;
    const target = activePlayers.find(p => p.id === id);
    if(target) {
      target.kicked = true;
      syncAll();
    }
  }

  function showGameOver(msg) {
    document.body.innerHTML = `
      <div class="screen fade-in" style="text-align:center; max-width:650px; margin:auto;">
        <div class="menu-logo" style="color:var(--neon-green); text-shadow:0 0 20px var(--neon-green-glow);">СЕССИЯ ЗАВЕРШЕНА</div>
        <div class="panel" style="margin-top:20px; border-color:var(--neon-green); text-align:left;">
          <div class="panel-title" style="color:var(--white);">Итоговый отчет главного ИИ:</div>
          <p style="font-size:0.95rem; line-height:1.6; color:var(--grey-text);">${msg}</p>
        </div>
        <button class="btn btn-accent" style="margin-top:25px;" onclick="window.location.reload()">Вернуться в главное меню</button>
      </div>
    `;
  }
